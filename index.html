from cryptography.hazmat.primitives.asymmetric.ed25519 import Ed25519PrivateKey
from dataclasses import dataclass, field
import hashlib, json, time

@dataclass
class AuditEntry:
    agent_id: str
    action: str
    payload_hash: str
    signature: bytes = field(default=None)
    timestamp: float = field(default_factory=time.time)

class Auditor:
    """Ed25519-signed immutable audit trail with
    entropy-based collusion detection."""

        self._key = Ed25519PrivateKey.generate()
        self._ledger: list[AuditEntry] = []

    def log_event(self, agent_id: str, action: str,
                 payload: dict) -> AuditEntry:
        digest = hashlib.sha256(
            json.dumps(payload, sort_keys=True).encode()
        ).hexdigest()
        entry = AuditEntry(agent_id, action, digest)
        entry.signature = self._key.sign(digest.encode())
        self._ledger.append(entry)
